📄 DocuCard — User Flow

Complete user journey from PDF upload to card generation — NotebookLM style 

flowchart TD
    START([🟢 User Opens DocuCard]) --> UPLOAD

    subgraph UPLOAD_PHASE["📤 Phase 1: Upload"]
        UPLOAD["Upload PDF(s)\nDrag & Drop or File Picker"] --> MULTI{Multiple PDFs?}
        MULTI -->|Yes| QUEUE["Add all to queue\neach processed independently"]
        MULTI -->|No| SINGLE[Single PDF ready]
        QUEUE --> PROCESS_BTN
        SINGLE --> PROCESS_BTN
        PROCESS_BTN["▶ Process All PDFs"]
    end

    PROCESS_BTN --> OCR_PHASE

    subgraph OCR_PHASE_BOX["🤖 Phase 2: AI Processing (Auto)"]
        OCR_PHASE["For each PDF..."] --> STEP_A
        STEP_A["Step A: PDF → Images\npdf.py / poppler 300 DPI"] --> STEP_B
        STEP_B["Step B: Gemini OCR\nExtract all text per page"] --> STEP_C
        STEP_C["Step C: Merge Page JSONs\ninto single document JSON"] --> STEP_D
        STEP_D["Step D: Auto-Classify\nGemini identifies PDF type"]
    end

    STEP_D --> CLASSIFY

    subgraph CLASSIFY_BOX["🏷️ Classification"]
        CLASSIFY{"What type\nis this PDF?"}
        CLASSIFY -->|"University Notice"| T1["🎓 PDF001"]
        CLASSIFY -->|"Government Circular"| T2["🏛️ PDF002"]
        CLASSIFY -->|"Job Circular"| T3["💼 PDF003"]
        CLASSIFY -->|"Financial Doc"| T4["🏦 PDF004"]
        CLASSIFY -->|"Meeting Minutes"| T5["📋 PDF005"]

        T1 --> RECOMMEND
        T2 --> RECOMMEND
        T3 --> RECOMMEND
        T4 --> RECOMMEND
        T5 --> RECOMMEND

        RECOMMEND["Map PDF Type → Best Card\n+ Score all 5 cards"]
    end

    RECOMMEND --> DASHBOARD

    subgraph DASHBOARD_PHASE["📊 Phase 3: Dashboard"]
        DASHBOARD["Dashboard View\nSidebar: All PDFs\nMain: Card selection"]
        DASHBOARD --> SHOW_CARDS

        SHOW_CARDS["Show ALL 5 Card Types\n⭐ Best match highlighted\nOthers still clickable"]

        SHOW_CARDS --> USER_CHOICE{"User selects\na card type"}

        USER_CHOICE -->|"Recommended ⭐"| HIGH_CONF["✅ High Match\nAll sections fill well"]
        USER_CHOICE -->|"Different Card"| CHECK_MATCH{"Match\nScore?"}

        CHECK_MATCH -->|"> 50%"| PARTIAL["⚠️ Partial Match\nSome sections empty"]
        CHECK_MATCH -->|"< 50%"| LOW["❌ Low Match\nWarning shown"]
        
        PARTIAL --> FILL
        LOW --> FILL
        HIGH_CONF --> FILL
    end

    subgraph FILL_PHASE["📋 Phase 4: Card View"]
        FILL["AI fills card sections\nfrom OCR data"] --> CARD_VIEW
        CARD_VIEW["Full Card Displayed\nPer-section confidence %"]
    end

    CARD_VIEW --> USER_ACTION

    subgraph ACTION_PHASE["✏️ Phase 5: User Actions"]
        USER_ACTION{"What next?"}
        USER_ACTION -->|Edit| EDIT["✏️ Edit section"]
        USER_ACTION -->|Regenerate| REGEN["🔄 Re-run AI"]
        USER_ACTION -->|Switch| SWITCH["🔀 Try another card"]
        USER_ACTION -->|Export| EXPORT["📥 PDF / JSON / HTML"]
        USER_ACTION -->|Save| SAVE["💾 Save card"]

        EDIT --> CARD_VIEW
        REGEN --> CARD_VIEW
        SWITCH --> SHOW_CARDS
    end

    EXPORT --> DONE
    SAVE --> DONE
    DONE(["🏁 Done"]) --> |"Upload more"| UPLOAD

🃏 Card Selection Decision Tree
What happens when user picks a card type for their PDF? 
flowchart TD
    PDF["📄 Any PDF Uploaded"] --> OCR["🤖 Gemini OCR\nExtracts all text"]
    OCR --> CLASSIFY["🏷️ Auto-Classify\nDetect PDF Type"]
    
    CLASSIFY --> SCORES["Calculate Match Scores\nfor all 5 card types"]
    
    SCORES --> SHOW["Show Dashboard\n5 cards with scores"]
    
    SHOW --> PICK{"User picks\na card"}
    
    PICK --> R1["🎓 Student Support"]
    PICK --> R2["💼 Job Eligibility"]
    PICK --> R3["🏛️ Gov Policy"]
    PICK --> R4["🏦 Financial Health"]
    PICK --> R5["📋 Meeting Tracker"]
    
    R1 --> CHECK{"Is this the\nrecommended card?"}
    R2 --> CHECK
    R3 --> CHECK
    R4 --> CHECK
    R5 --> CHECK
    
    CHECK -->|"Yes ⭐"| PERFECT["✅ Perfect Match\n• All sections filled\n• High confidence 80-98%\n• No warnings"]
    CHECK -->|"No"| SCORE_CHECK{"Match score?"}
    
    SCORE_CHECK -->|"> 50%"| PARTIAL["⚠️ Partial Match\n• Some sections filled\n• Mid confidence 30-70%\n• Yellow warning banner"]
    SCORE_CHECK -->|"< 50%"| MISMATCH["❌ Mismatch\n• Most sections empty\n• Low confidence 5-25%\n• Red warning banner\n• Suggestion to switch"]
    
    PERFECT --> EDIT["✏️ User can edit,\nregenerate, export"]
    PARTIAL --> EDIT
    MISMATCH --> SWITCH{"User can\nswitch card\nor proceed"}
    
    SWITCH -->|"Switch"| SHOW
    SWITCH -->|"Proceed anyway"| EDIT

    style MISMATCH fill:#2a0a0a,stroke:#ef4444,color:#ef4444

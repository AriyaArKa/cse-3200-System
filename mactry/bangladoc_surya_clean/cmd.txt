# documents  -> each uploaded file metadata and status

# Sample checks in Query Tool

SELECT id, filename, doc_id, status, total_pages, created_at FROM documents ORDER BY created_at DESC;
SELECT id, document_id, status, error_msg, started_at, completed_at FROM ocr_jobs ORDER BY started_at DESC NULLS LAST;
SELECT document_id, page_number, engine, confidence_score, verified FROM pages ORDER BY document_id, page_number;

11) Troubleshooting
------------------
# If Surya import fails
source /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean/backend/venv/bin/activate
python -m pip install --upgrade surya-ocr

# If Surya shows: "SuryaDecoderConfig has no attribute pad_token_id"
# (caused by incompatible transformers 5.x)
python -m pip install "transformers<5"
python -m pip install -e "./backend[dev]"
python -c "from surya.foundation import FoundationPredictor; from surya.recognition import RecognitionPredictor; f=FoundationPredictor(); RecognitionPredictor(f); print('surya predictors ok')"

# If Ollama connection fails
ollama serve
ollama list

# If package command bangladoc not found
cd /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean
source backend/venv/bin/activate
python -m pip install -e "./backend[dev]"

# If API says: address already in use (port 8000)
# 1) Check whether API is already running:
python - <<'PY'
import requests
try:
	r = requests.get('http://127.0.0.1:8000/health', timeout=3)
	print('API already running:', r.status_code, r.text)
except Exception as e:
	print('API not reachable:', e)
PY
# 2) If already running, do NOT start another uvicorn process.

# If Celery crashes with WorkerLostError / SIGABRT on macOS MPS
# Use non-fork worker pool:
python -m celery -A bangladoc_ocr.celery_app:celery_app worker --loglevel=info --pool=solo -n worker1@%h

# If macOS shows "Python quit unexpectedly" during worker run
# 1) Stop all old celery workers first (duplicate workers cause instability)
pkill -f "celery -A bangladoc_ocr.celery_app:celery_app worker" || true
pkill -f "python -m celery -A bangladoc_ocr.celery_app:celery_app worker" || true

# 2) Confirm no worker remains
pgrep -fl "celery" || true

# 3) Start one stable worker only (single process, no forking)
python -m celery -A bangladoc_ocr.celery_app:celery_app worker --loglevel=info --pool=solo -n worker1@%h

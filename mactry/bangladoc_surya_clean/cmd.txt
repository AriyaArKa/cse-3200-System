1) First-time setup from scratch
--------------------------------
cd /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean

# 1.1 Environment file
cp backend/.env.example backend/.env

# 1.1.1 OCR flow control (edit backend/.env)
# SURYA_ENABLED=true
# true  => surya -> ollama -> gemini -> easyocr
# false => ollama -> gemini -> easyocr
# DATA_DIR=../data            # outputs save in bangladoc_surya_clean/data

# 1.2 Create virtual environment
python3 -m venv backend/venv
source backend/venv/bin/activate
python -m pip install --upgrade pip setuptools wheel

# 1.3 Install backend package and dev dependencies
python -m pip install -e "./backend[dev]"

# 1.4 Apple Silicon fallback for PyTorch/Surya
export PYTORCH_ENABLE_MPS_FALLBACK=1

# 1.5 Pull required Ollama models
ollama list | grep "qwen2.5vl:7b" || ollama pull qwen2.5vl:7b
ollama list | grep "moondream2" || ollama pull moondream2

# 1.6 Verify Surya import
python -c "from surya.recognition import RecognitionPredictor; print('surya ok')"

# 1.7 Start postgres container
cd /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean
docker compose up -d postgres

# 1.8 Quick import health check
cd backend
source venv/bin/activate
python -c "from bangladoc_ocr.pipeline import process_pdf; print('pipeline import ok')"

8) Troubleshooting
------------------
# If Surya import fails
source /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean/backend/venv/bin/activate
python -m pip install --upgrade surya-ocr

# If Surya shows: "SuryaDecoderConfig has no attribute pad_token_id"
# (caused by incompatible transformers 5.x)
python -m pip install "transformers<5"
python -m pip install -e "./backend[dev]"
python -c "from surya.foundation import FoundationPredictor; from surya.recognition import RecognitionPredictor; f=FoundationPredictor(); RecognitionPredictor(f); print('surya predictors ok')"

# If Ollama connection fails
ollama serve
ollama list

# If package command bangladoc not found
cd /Users/arkabrajaprasadnath/3-2/System\ PRoject/cse-3200-System/mactry/bangladoc_surya_clean
source backend/venv/bin/activate
python -m pip install -e "./backend[dev]"
